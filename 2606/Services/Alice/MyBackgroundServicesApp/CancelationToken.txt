Базовые принципы плавной остановки
Чтобы сервисы останавливались предсказуемо, нужно соблюдать три принципа:

Реагировать на CancellationToken — это стандарт .NET для отмены операций.

Завершать текущую работу перед выходом — не прерывать операции на середине.

Логировать этапы остановки — для диагностики и мониторинга.

Вариант 1. Общий базовый класс (рекомендуемый)
Создайте абстрактный класс, который задаёт структуру, но оставляет логику подклассам:

csharp
public abstract class GracefulBackgroundService : BackgroundService
{
    protected abstract Task ExecuteWorkAsync(CancellationToken stoppingToken);

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("Service started...");

        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                await ExecuteWorkAsync(stoppingToken);
            }
            catch (OperationCanceledException) when (stoppingToken.IsCancellationRequested)
            {
                _logger.LogInformation("Operation cancelled due to service shutdown.");
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error during service execution.");
                await Task.Delay(TimeSpan.FromSeconds(5), stoppingToken); // пауза перед повторной попыткой
            }
        }

        _logger.LogInformation("Service is shutting down gracefully...");
    }
}

{
    Running,
    FinishingCurrentWork,
    ShuttingDown
}

private ShutdownPhase _shutdownPhase = ShutdownPhase.Running;

protected override async Task ExecuteAsync(CancellationToken stoppingToken)
{
    while (_shutdownPhase != ShutdownPhase.ShuttingDown)
    {
        if (stoppingToken.IsCancellationRequested && _shutdownPhase == ShutdownPhase.Running)
        {
            _shutdownPhase = ShutdownPhase.FinishingCurrentWork;
            _logger.LogInformation("Shutdown initiated: finishing current work...");
        }

        if (_shutdownPhase == ShutdownPhase.Running)
        {
            await DoWorkAsync(stoppingToken);
        }
        else if (_shutdownPhase == ShutdownPhase.FinishingCurrentWork)
        {
            // Даём время на завершение текущей работы
            await Task.Delay(TimeSpan.FromSeconds(1), stoppingToken);
            _shutdownPhase = ShutdownPhase.ShuttingDown;
        }
    }

    _logger.LogInformation("Service stopped.");
}

{
    /// <summary>Время на завершение текущей работы после сигнала остановки</summary>
    public TimeSpan GracePeriod { get; set; } = TimeSpan.FromSeconds(30);

    /// <summary>Разрешить завершение текущей операции перед остановкой</summary>
    public bool AllowCurrentWorkCompletion { get; set; } = true;

    /// <summary>Задержка между попытками при ошибках</summary>
    public TimeSpan RetryDelay { get; set; } = TimeSpan.FromSeconds(10);

    /// <summary>Таймаут для отдельных операций</summary>
    public TimeSpan OperationTimeout { get; set; } = TimeSpan.FromMinutes(1);
}

public abstract class ConfigurableBackgroundService : BackgroundService
{
    private readonly GracefulShutdownOptions _options;
    private readonly ILogger _logger;

    protected ConfigurableBackgroundService(GracefulShutdownOptions options, ILogger logger)
    {
        _options = options;
        _logger = logger;
    }

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("Service started with configuration: GracePeriod={GracePeriod}, AllowCurrentWorkCompletion={AllowCurrentWorkCompletion}",
            _options.GracePeriod, _options.AllowCurrentWorkCompletion);

        try
        {
            while (!stoppingToken.IsCancellationRequested)
            {
                try
                {
                    // Бизнес‑логика конкретного сервиса
                    await PerformWorkAsync(stoppingToken);
                }
                catch (OperationCanceledException) when (stoppingToken.IsCancellationRequested)
                {
                    _logger.LogWarning("Shutdown signal received. Initiating graceful shutdown...");
                    if (_options.AllowCurrentWorkCompletion)
            {
                _logger.LogInformation("Allowing current work to complete within {GracePeriod}...", _options.GracePeriod);
                await Task.Delay(_options.GracePeriod, stoppingToken);
            }
            break;
        }
        catch (HttpRequestException ex)
        {
            _logger.LogError(ex, "HTTP request failed. Retrying after {RetryDelay}...", _options.RetryDelay);
            await Task.Delay(_options.RetryDelay, stoppingToken);
        }
        catch (TimeoutException ex)
        {
            _logger.LogError(ex, "Operation timed out. Retrying after {RetryDelay}...", _options.RetryDelay);
            await Task.Delay(_options.RetryDelay, stoppingToken);
        }
        catch (IOException ex)
        {
            _logger.LogError(ex, "I/O error occurred. Retrying after {RetryDelay}...", _options.RetryDelay);
            await Task.Delay(_options.RetryDelay, stoppingToken);
        }
        catch (Exception ex) when (!stoppingToken.IsCancellationRequested)
        {
            _logger.LogCritical(ex, "Unexpected error. Retrying after {RetryDelay}...", _options.RetryDelay);
            await Task.Delay(_options.RetryDelay, stoppingToken);
        }
    }
}
finally
{
    _logger.LogInformation("Service is shutting down gracefully...");
}
}

/// <summary>
/// Бизнес‑логика конкретного сервиса — должна быть реализована в наследниках
/// </summary>
protected abstract Task PerformWorkAsync(CancellationToken stoppingToken);
}

Для сервиса отправки команд (аналог кода № 1):

    AllowCurrentWorkCompletion = true,                  // дать закончить текущую отправку

    OperationTimeout = TimeSpan.FromSeconds(30)      // таймаут на операцию 30 с

Как это работает
Конфигурация задаётся один раз при создании сервиса — можно настроить под специфику каждой задачи.

Сигнал остановки (Ctrl+C) перехватывается через stoppingToken.

Плавное завершение:

если AllowCurrentWorkCompletion = true, даётся время GracePeriod на завершение текущей операции;

иначе — немедленный выход.

Обработка ошибок:

конкретные исключения (HttpRequestException, TimeoutException и т. д.) обрабатываются с задержкой RetryDelay;

непредвиденные ошибки тоже не приводят к падению — сервис ждёт и повторяет.

Логирование фиксирует все этапы: старт, получение сигнала, завершение, остановку.

Преимущества подхода
Гибкость: параметры можно менять без правки кода — достаточно обновить конфигурацию.

Унификация: все сервисы наследуют один и тот же механизм остановки.

Наглядность: логика чёткая, исключения разнесены по типам.

Устойчивость: сервис не падает при временных ошибках, а повторяет попытки.

Диагностичность: логи показывают, что происходит на каждом этапе.

Масштабируемость: легко добавить новые параметры или типы исключений.

Рекомендации по настройке параметров
GracePeriod:

быстрые операции (отправка команд) — 5–15 с;

тяжёлые задачи (обработка батчей) — 30–120 с.

RetryDelay:

сетевые ошибки — 5–10 с;

ошибки БД — 10–30 с;

критические ошибки — до 60 с.

OperationTimeout:

установите чуть больше, чем ожидаемое время выполнения операции;

для долгих задач — до нескольких минут.
