// Worker.cs (пример использования TopologyLoader и NodeRegistry)
using FractalCellCore.Core.Interfaces;
using FractalCellCore.Core.Topology;
using Microsoft.Extensions.Hosting;
using Microsoft.Extensions.Logging;

namespace FractalCellApp;

public class Worker : BackgroundService
{
    private readonly ILogger<Worker> _logger;
    private readonly TopologyLoader _topologyLoader;
    private readonly INodeRegistry _nodeRegistry;
    private INode? _root;

    public Worker(ILogger<Worker> logger, TopologyLoader topologyLoader, INodeRegistry nodeRegistry)
    {
        _logger = logger;
        _topologyLoader = topologyLoader;
        _nodeRegistry = nodeRegistry;
    }

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("Worker starting: loading topology...");

        var json = await File.ReadAllTextAsync("topology.example.json", stoppingToken);
        _root = await _topologyLoader.LoadFromJsonAsync(json, stoppingToken);

        // Рекурсивная регистрация всех узлов в реестре
        await RegisterNodesRecursiveAsync(_root, stoppingToken);

        // Запускаем корневой оркестратор
        await _root.StartAsync(stoppingToken);

        _logger.LogInformation("Topology loaded and started. Worker idle until cancellation.");

        try
        {
            await Task.Delay(-1, stoppingToken);
        }
        catch (OperationCanceledException)
        {
            _logger.LogInformation("Worker stopping due to cancellation");
        }
        finally
        {
            if (_root != null)
            {
                await _root.StopAsync(CancellationToken.None);
            }
        }
    }

    private async Task RegisterNodesRecursiveAsync(INode node, CancellationToken ct)
    {
        await _nodeRegistry.RegisterAsync(node, ct);

        if (node is ICompositeNode comp)
        {
            foreach (var child in comp.Children)
            {
                await RegisterNodesRecursiveAsync(child, ct);
            }
        }
    }
}
